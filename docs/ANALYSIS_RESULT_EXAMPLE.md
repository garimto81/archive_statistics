```sql
-- íŒŒì¼ë³„ ë¶„ì„ í˜„í™© ì¡°íšŒ
SELECT
    ph.file_name,
    fs.duration_seconds,
    MAX(ph.time_end_s) as max_timecode,
    COUNT(*) as hand_count,
    ROUND(MAX(ph.time_end_s) / fs.duration_seconds * 100, 1) as coverage_percent,
    CASE
        WHEN MAX(ph.time_end_s) >= fs.duration_seconds * 0.9 THEN 'COMPLETE'
        WHEN COUNT(*) > 0 THEN 'IN_PROGRESS'
        ELSE 'NOT_STARTED'
    END as status
FROM poker_hands ph
LEFT JOIN file_stats fs ON ph.file_name = fs.name
GROUP BY ph.file_name, fs.duration_seconds
ORDER BY coverage_percent DESC;
```
